// [dbdiagram.io 전용 DBML 포맷 - 최종 7개 테이블 모델]
// 확장성(Scalability)과 논리적 완전성(Logical Completeness)을 고려한 설계

// [1] 사용자 (Users)
Table users {
  id int [pk, increment]
  login_id varchar [unique, not null]
  email varchar [unique, not null, note: '비밀번호 찾기 등 인증용']
  password varchar [not null]
  nickname varchar [not null]
  identity_info varchar [note: '대학/전공 등 소속 증빙 정보']
  status varchar [default: 'ACTIVE', note: 'ACTIVE, SUSPENDED, DELETED']
  created_at timestamp [default: `now()`]
  updated_at timestamp [default: `now()`]
  deleted_at timestamp [note: 'Soft delete용']
}

// [2] 학습 자료 (Materials)
Table materials {
  id int [pk, increment]
  user_id int [ref: > users.id]
  category varchar [note: '족보, 요약본, 강의록 등 자료 유형']
  title varchar [not null]
  content text
  file_url varchar
  view_count int [default: 0]
  status varchar [default: 'PUBLISHED', note: 'DRAFT, PUBLISHED, ARCHIVED']
  created_at timestamp [default: `now()`]
  updated_at timestamp [default: `now()`]
  deleted_at timestamp [note: 'Soft delete용']
}

// [3] 통합 상호작용 (Interactions)
Table interactions {
  id int [pk, increment]
  user_id int [ref: > users.id]
  material_id int [ref: > materials.id]
  parent_id int [ref: > interactions.id, note: '대댓글(Nested Comment) 구현 등 계층 구조를 위한 셀프 참조']
  content text [note: 'LIKE 타입의 경우 null 가능']
  type varchar [not null, note: 'COMMENT, NOTE, LIKE 등 이벤트 타입 분류']
  is_public boolean [default: true, note: '오답노트 비공개 등 설정 여부']
  created_at timestamp [default: `now()`]
  updated_at timestamp [default: `now()`]
  deleted_at timestamp [note: 'Soft delete용']
}

// [4] 스터디/모임 (Groups)
Table groups {
  id int [pk, increment]
  unique_code varchar [unique, not null, note: '초대용 고유 코드']
  name varchar [not null]
  description text
  max_members int [default: 50]
  creator_id int [ref: > users.id]
  status varchar [default: 'ACTIVE', note: 'ACTIVE, CLOSED']
  created_at timestamp [default: `now()`]
  updated_at timestamp [default: `now()`]
  deleted_at timestamp [note: 'Soft delete용']
}

// [5] 모임 멤버 (Members)
Table members {
  group_id int [ref: > groups.id]
  user_id int [ref: > users.id]
  role varchar [default: 'MEMBER', note: 'OWNER, ADMIN, MEMBER 등 권한 등급']
  status varchar [default: 'ACTIVE', note: 'ACTIVE, PENDING, BANNED 등 상태값']
  joined_at timestamp [default: `now()`]
  deleted_at timestamp [note: 'Soft delete 탈퇴 처리용']
  
  indexes {
    (group_id, user_id) [pk]
  }
}

// [6] 모임 채팅 (Chats)
Table chats {
  id int [pk, increment]
  group_id int [ref: > groups.id]
  user_id int [ref: > users.id]
  message text [not null]
  message_type varchar [default: 'TEXT', note: 'TEXT, IMAGE, FILE 등 메시지 포맷']
  sent_at timestamp [default: `now()`]
  deleted_at timestamp [note: '메시지 회수 등 처리(Soft delete)']
}

// [7] 개인 및 모임 일정 (Schedules)
Table schedules {
  id int [pk, increment]
  user_id int [ref: > users.id, note: '일정 생성자(소유자)']
  group_id int [ref: > groups.id, note: 'null이면 개인 일정, 값이 있으면 모임(그룹) 공통 일정']
  title varchar [not null]
  description text
  location varchar [note: '온/오프라인 모임 장소']
  is_all_day boolean [default: false]
  start_at datetime [not null]
  end_at datetime [not null]
  created_at timestamp [default: `now()`]
  updated_at timestamp [default: `now()`]
  deleted_at timestamp [note: 'Soft delete용']
}
